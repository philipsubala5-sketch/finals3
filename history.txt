import os
import json
import sys
import math
import random
import threading
import time


# ===============================================
# SYSTEM INITIALIZATION & HARDWARE PROFILE LOADING
# ===============================================

CONFIG_FILE = "hardware_config.json"

if not os.path.exists(CONFIG_FILE):
    print("[FATAL ERROR] Hardware configuration not found!")
    print("Place hardware_config.json in the same folder as main.py")
    sys.exit(1)

with open(CONFIG_FILE, "r") as f:
    CONFIG = json.load(f)

print(
    f"[SYSTEM] Avionics Bus initialized for Signature: "
    f"{CONFIG.get('signature_hash', 'UNKNOWN')}"
)


# ===============================================
# MODULE 1: TELEMETRY BUFFER
# ===============================================

class TelemetryBuffer:

    def __init__(self, channel_id: str, frame_buffer=None):
        self.channel_id = channel_id
        self.frame_buffer = frame_buffer if frame_buffer else []
        self.dropped_packets = 0


    def push_frame(self, timestamp: float, voltage_level: float):

        packet = {
            "ts": round(timestamp, 4),
            "val": round(voltage_level, 4),
            "status": "NOMINAL"
        }

        self.frame_buffer.append(packet)

        max_capacity = CONFIG.get(
            "buffer_overflow_threshold",
            25
        )

        if len(self.frame_buffer) > max_capacity:
            raise MemoryError(
                f"[{CONFIG['signature_hash']}] "
                f"Hardware Buffer Overflow on "
                f"Channel {self.channel_id}"
            )


    def flush_buffer(self):

        count = len(self.frame_buffer)
        self.frame_buffer.clear()
        return count



# ===============================================
# MODULE 2: HARDWARE PERIPHERALS
# ===============================================

class AvionicsPeripheral:

    def __init__(self, pin_number, peripheral_name):

        self.pin_number = pin_number
        self.peripheral_name = peripheral_name
        self.buffer = TelemetryBuffer(
            f"CH_{pin_number}_{peripheral_name}"
        )
        self.is_active = True


    def poll_raw_voltage(self):
        return 0.0



class LiDARPeripheral(AvionicsPeripheral):

    def poll_raw_voltage(self):

        voltage = 2.5 + math.sin(time.time()) * 0.5

        self.buffer.push_frame(
            time.time(),
            voltage
        )

        return voltage



class GPSPeripheral(AvionicsPeripheral):

    def poll_raw_voltage(self):

        voltage = 3.3 + random.uniform(
            -0.05,
            0.05
        )

        self.buffer.push_frame(
            time.time(),
            voltage
        )

        return voltage



class IMUPeripheral(AvionicsPeripheral):

    def poll_raw_voltage(self):

        drift = CONFIG.get(
            "imu_voltage_drift",
            1.0
        )

        voltage = 1.8 * (drift / 5.0)

        self.buffer.push_frame(
            time.time(),
            voltage
        )

        return voltage



# ===============================================
# MODULE 3: AVIONICS BUS CONTROLLER
# ===============================================

class AvionicsBusMaster:

    def __init__(self):

        self.peripherals = [
            LiDARPeripheral(
                10,
                "LIDAR_FRONT"
            ),
            GPSPeripheral(
                12,
                "GPS_MAIN"
            ),
            IMUPeripheral(
                14,
                "IMU_NAV"
            )
        ]

        self.active_bus_register = {}
        self.total_cycles_executed = 0
        self.lock = threading.Lock()



    def _async_polling_worker(
        self,
        peripheral,
        cycles
    ):

        for _ in range(cycles):

            raw_signal = peripheral.poll_raw_voltage()

            time.sleep(
                CONFIG.get(
                    "bus_race_delay_sec",
                    0.002
                )
            )

            normalized = (
                raw_signal * 5.0
            ) / 3.3


            with self.lock:

                self.active_bus_register[
                    peripheral.peripheral_name
                ] = {

                    "packets_received": 
                        self.active_bus_register.get(
                            peripheral.peripheral_name,
                            {}
                        ).get(
                            "packets_received",
                            0
                        ) + 1,

                    "latest_voltage":
                        round(normalized, 4),

                    "status":
                        "SYNCED"
                }


                self.total_cycles_executed += 1



    def begin_flight_telemetry_loop(
        self,
        cycles_per_peripheral=10
    ):

        print(
            "--- Launching Avionics Bus Master ---"
        )


        threads = []


        for peripheral in self.peripherals:

            thread = threading.Thread(
                target=self._async_polling_worker,
                args=(
                    peripheral,
                    cycles_per_peripheral
                )
            )

            threads.append(thread)
            thread.start()


        for thread in threads:
            thread.join()


        print(
            "--- Telemetry Loop Terminated ---"
        )

        print(
            f"Total Register Writes: "
            f"{self.total_cycles_executed}"
        )


        self.dump_bus_diagnostics()



    def dump_bus_diagnostics(self):

        print(
            "\n--- FINAL AVIONICS BUS REGISTER STATE ---"
        )

        for channel, data in self.active_bus_register.items():

            print(
                f"{channel}: {data}"
            )



# ===============================================
# PROGRAM START
# ===============================================

if __name__ == "__main__":

    bus_master = AvionicsBusMaster()

    try:

        bus_master.begin_flight_telemetry_loop(
            cycles_per_peripheral=12
        )

    except Exception as e:

        print(
            "[CRITICAL SYSTEM FAULT]"
        )

        print(
            f"{type(e).__name__}: {e}"
        )

        sys.exit(1)
